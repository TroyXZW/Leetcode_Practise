# 最小覆盖字串 p87-算法小抄

```
#!/usr/bin/env python
# -*- coding:utf-8 -*-

def minWindow(s, t):
    need = {}
    window = {}
    for i in t:
        n = 1 if need.get(i, 0) == 0 else need.get(i, 0) + 1
        need[i] = n
    left, right = 0, 0
    valid = 0
    start = 0
    length = float('inf')
    while right < len(s):
        c = s[right]
        right += 1
        if need.get(c, -1) != -1:
            n = 1 if window.get(c, 0) == 0 else window.get(c, 0) + 1
            window[c] = n
            if window[c] == need[c]:
                valid += 1
        while valid == len(need):
            if right - left < length:  # 为了找最小的length
                start = left
                length = right - left
            d = s[left]
            left += 1
            if need.get(d, -1) != -1:
                if window[d] == need[d]:
                    valid -= 1
                window[d] -= 1
    return "" if length == float('inf') else s[start:start + length]


if __name__ == "__main__":
    res = minWindow("adbecfebanc", "abc")
    print(res)

```
