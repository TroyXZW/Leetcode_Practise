# 最小覆盖字串 p87-算法小抄

```
#!/usr/bin/env python
# -*- coding:utf-8 -*-

def minWindow(s, t):
    need = {}
    window = {}
    for i in t:
        n = 1 if need.get(i, 0) == 0 else need.get(i, 0) + 1
        need[i] = n
    left, right = 0, 0
    valid = 0
    start = 0
    length = float('inf')
    while right < len(s):
        c = s[right]
        right += 1
        if need.get(c, -1) != -1:
            n = 1 if window.get(c, 0) == 0 else window.get(c, 0) + 1
            window[c] = n
            if window[c] == need[c]:
                valid += 1
        while valid == len(need):
            if right - left < length:  # 为了找最小的length
                start = left
                length = right - left
            d = s[left]
            left += 1
            if need.get(d, -1) != -1:
                if window[d] == need[d]:
                    valid -= 1
                window[d] -= 1
    return "" if length == float('inf') else s[start:start + length]


if __name__ == "__main__":
    res = minWindow("adbecfebanc", "abc")
    print(res)

```


# 字符串排列 p91-算法小抄

#!/usr/bin/env python
# -*- coding:utf-8 -*-

def checkInclusion(s, t):
    need = {}
    window = {}
    for i in t:
        n = 1 if need.get(i, 0) == 0 else need.get(i, 0) + 1
        need[i] = n
    left, right = 0, 0
    valid = 0
    while right < len(s):
        c1 = s[right]
        right += 1
        if c1 in need:
            n = 1 if window.get(c1, 0) == 0 else window.get(c1, 0) + 1
            window[c1] = n
            if window[c1] == need[c1]:
                valid += 1
        while right - left >= len(t):  # right - left == len(t)可不？
            if valid == len(need):
                return True

            c2 = s[left]
            left += 1
            if c2 in need:
                if window[c2] == need[c2]:
                    valid -= 1
                window[c2] -= 1
    return False


if __name__ == "__main__":
    res = checkInclusion("helloworld", "oow")
    print(res)

# 找所有字母异位词 p93-算法小抄

```
#!/usr/bin/env python
# -*- coding:utf-8 -*-

def finAnagrams(s, t):
    need = {}
    window = {}
    for i in t:
        n = 1 if need.get(i, 0) == 0 else need.get(i, 0) + 1
        need[i] = n
    left, right = 0, 0
    valid = 0
    result = []
    while right < len(s):
        c1 = s[right]
        right += 1
        if c1 in need:
            n = 1 if window.get(c1, 0) == 0 else window.get(c1, 0) + 1
            window[c1] = n
            if window[c1] == need[c1]:
                valid += 1
        while right - left == len(t):
            if valid == len(need):
                result.append(left)

            c2 = s[left]
            left += 1
            if c2 in need:
                if window[c2] == need[c2]:
                    valid -= 1
                window[c2] -= 1
    return result


if __name__ == "__main__":
    res = finAnagrams("cbaebabacd", "abc")
    print(res)

```


# 最长无重复子串 p94-算法小抄

```
#!/usr/bin/env python
# -*- coding:utf-8 -*-

def lengthOfLongestSubstring(s):
    window = {}
    left, right = 0, 0
    max_len = 0
    while right < len(s):
        c1 = s[right]
        right += 1
        n = 1 if window.get(c1, 0) == 0 else window.get(c1, 0) + 1
        window[c1] = n
        while window[c1] != 1:
            c2 = s[left]
            left += 1
            window[c2] -= 1
        max_len = max(max_len, right - left)
    return max_len


if __name__ == "__main__":
    res = lengthOfLongestSubstring("aabab")
    print(res)

```
