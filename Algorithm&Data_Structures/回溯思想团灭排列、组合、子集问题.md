def backtrack(nums, start, track):
    res.append(track[:])  # 一定要用列表的引用，否者后面更新track会改变res中元素的值, res.append(track)报错
    for i in range(start, len(nums)):
        track.append(nums[i])
        backtrack(nums, i + 1, track)
        track.pop()


def subset(nums):
    track = []
    backtrack(nums, 0, track)
